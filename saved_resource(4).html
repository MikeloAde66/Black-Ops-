Node 2: Calculate Atmospheric State

javascript
// Function Node
const missions = await $workflow.execute('Get Active Missions');
const currentTime = new Date();

// Calculate mission pressure
let totalPressure = 0;
let teamSyncScore = 0.8; // From agent interactions
let complexityLevel = 0;

missions.forEach(mission => {
  const deadline = new Date(mission.deadline);
  const timeRemaining = (deadline - currentTime) / (1000 * 60 * 60); // hours
  const originalTime = mission.estimated_hours;
  const deadlineProximity = 1 - (timeRemaining / originalTime);
  
  totalPressure += Math.max(0, deadlineProximity);
  complexityLevel += mission.complexity || 0.5;
});

const avgPressure = totalPressure / (missions.length || 1);
const avgComplexity = complexityLevel / (missions.length || 1);

// Determine atmospheric state
let atmosphericState = 'flow';
let intensity = 0.5;

// TENSION: High pressure, approaching deadline
if (avgPressure > 0.7 && teamSyncScore < 0.9) {
  atmosphericState = 'tension';
  intensity = 0.7 + (avgPressure * 0.3);
}

// FLOW: High sync, good progress
else if (teamSyncScore > 0.85 && avgComplexity > 0.6 && avgPressure < 0.7) {
  atmosphericState = 'flow';
  intensity = teamSyncScore;
}

// CHAOS: Multiple high-pressure situations
else if (avgPressure > 0.8 && missions.length > 2) {
  atmosphericState = 'chaos';
  intensity = Math.min(1.0, avgPressure + (missions.length * 0.1));
}

// CELEBRATION: Low pressure, high success
else if (avgPressure < 0.3 && teamSyncScore > 0.9) {
  atmosphericState = 'celebration';
  intensity = teamSyncScore;
}

// UNCERTAINTY: Default or ambiguous
else {
  atmosphericState = 'uncertainty';
  intensity = 0.5;
}

return {
  json: {
    state: atmosphericState,
    intensity: intensity,
    metrics: {
      avgPressure,
      teamSyncScore,
      avgComplexity,
      activeMissions: missions.length
    },
    timestamp: currentTime.toISOString()
  }
};

javascript
// Function Node: Analyze Mission
const mission = $input.item.json;

// Extract key signals
const signals = {
  keywords: mission.description.toLowerCase(),
  sector: mission.sector || 'unknown',
  budget: mission.budget || 0,
  timeline: mission.timeline_days || 90,
  complexity: mission.complexity || 0.5
};

// Scoring system for each agent
const scores = {
  veronica: 0,
  cipher: 0,
  maven: 0,
  phoenix: 0
};

// VERONICA signals
const veronicaKeywords = ['luxury', 'brand', 'relationship', 'network', 'high-net-worth', 'exclusive', 'fashion', 'premium'];
veronicaKeywords.forEach(keyword => {
  if (signals.keywords.includes(keyword)) scores.veronica += 0.2;
});
if (signals.budget > 100000) scores.veronica += 0.3;
if (['luxury', 'fashion', 'lifestyle'].includes(signals.sector)) scores.veronica += 0.5;

// CIPHER signals
const cipherKeywords = ['tech', 'software', 'data', 'security', 'infrastructure', 'system', 'platform', 'api'];
cipherKeywords.forEach(keyword => {
  if (signals.keywords.includes(keyword)) scores.cipher += 0.2;
});
if (['technology', 'saas', 'fintech'].includes(signals.sector)) scores.cipher += 0.5;
if (signals.complexity > 0.7) scores.cipher += 0.3;

// MAVEN signals
const mavenKeywords = ['content', 'creative', 'brand', 'social', 'viral', 'campaign', 'influencer', 'media'];
mavenKeywords.forEach(keyword => {
  if (signals.keywords.includes(keyword)) scores.maven += 0.2;
});
if (['media', 'entertainment', 'creative'].includes(signals.sector)) scores.maven += 0.5;
if (signals.timeline < 30) scores.maven += 0.3; // Fast turnaround

// PHOENIX signals
const phoenixKeywords = ['turnaround', 'restructure', 'bankruptcy', 'rescue', 'failing', 'distressed', 'recovery'];
phoenixKeywords.forEach(keyword => {
  if (signals.keywords.includes(keyword)) scores.phoenix += 0.3;
});
if (['all'].includes(signals.sector)) scores.phoenix += 0.3; // She's versatile
if (signals.complexity > 0.8) scores.phoenix += 0.4; // Complex situations

// Determine primary agent
const primaryAgent = Object.keys(scores).reduce((a, b) => 
  scores[a] > scores[b] ? a : b
);

// Determine if collaboration needed (scores within 0.3 of each other)
const collaborators = Object.keys(scores).filter(agent => 
  agent !== primaryAgent && Math.abs(scores[agent] - scores[primaryAgent]) < 0.3
);

return {
  json: {
    mission_id: mission.id || Date.now(),
    mission: mission,
    primary_agent: primaryAgent,
    collaborators: collaborators,
    scores: scores,
    atmospheric_state: process.env.CURRENT_ATMOSPHERE || 'flow',
    assigned_at: new Date().toISOString()
  }
};

// Function Node: Post-Process Veronica's Response

  cipher: /cipher|technical|data|infrastructure|system/i.test(response),
  maven: /maven|creative|brand|content|social/i.test(response),
  phoenix: /phoenix|strategy|plan|financial|turnaround/i.test(response)
};

// Detect if task agents needed
const taskAgentNeeds = {
  email: /email|outreach|contact|reach out/i.test(response),
  research: /research|investigate|find out|look into/i.test(response),
  relationship_mapping: /network|connections|who knows/i.test(response)
};

// Extract action items
const actionItems = [];
if (collaborationSignals.cipher || collaborationSignals.maven || collaborationSignals.phoenix) {
  actionItems.push({
    type: 'collaborate',
    with: Object.keys(collaborationSignals).filter(k => collaborationSignals[k])
  });

  if (taskAgentNeeds[taskType]) {
    actionItems.push({
      type: 'spawn_task_agent',

      created_by: 'veronica'
    });
  }
});

return {
  json: {
    agent: 'veronica',
    response: response,
    mission_id: mission.id,
    action_items: actionItems,
    collaboration_needed: actionItems.some(a => a.type === 'collaborate'),
    task_agents_needed: actionItems.filter(a => a.type === 'spawn_task_agent'),
    timestamp: new Date().toISOString()
  }

// Create task agent based on request
const request = $input.item.json;
const template = taskAgentTemplates[request.task_type];

const taskAgent = {
  id: `task_${Date.now()}`,
  name: `${template.name} (${request.created_by})`,
  type: request.task_type,
  created_by: request.created_by,
  created_at: new Date().toISOString(),
  capabilities: template.capabilities,
  tools: template.tools,
  training_data: {
    learned_from: request.created_by,
    communication_style: await getAgentStyle(request.created_by),
    domain_knowledge: request.domain_context
  },
  status: 'active',
  missions_completed: 0
};

return { json: taskAgent };

Common Postgres Nodes

sql
-- Log Mission Assignment
INSERT INTO missions (
  mission_name,
  assigned_to,
  atmospheric_state,
  status
) VALUES (
  '{{ $json.mission.name }}',
  '{{ $json.primary_agent }}',
  '{{ $json.atmospheric_state }}',
  'in_progress'
) RETURNING id;

-- Log Agent Interaction
INSERT INTO agent_interactions (
  agent_name,
  interaction_type,
  content,
  context,
  atmospheric_state
) VALUES (
  '{{ $json.agent }}',
  'mission_response',
  '{{ $json.response }}',
  '{{ JSON.stringify($json.mission) }}',
  '{{ $json.atmospheric_state }}'
);

-- Register Task Agent
INSERT INTO task_agents (
  name,
  type,
  created_by,
  capabilities,
  status
) VALUES (
  '{{ $json.name }}',
  '{{ $json.type }}',
  '{{ $json.created_by }}',
  '{{ JSON.stringify($json.capabilities) }}',
  'active'
) RETURNING id;

-- Get Active Missions
SELECT * FROM missions 
WHERE status = 'in_progress'
AND completed_at IS NULL
ORDER BY created_at DESC;

-- Get Agent Performance
SELECT 
  agent_name,
  COUNT(*) as total_interactions,
  AVG(CASE WHEN atmospheric_state = 'flow' THEN 1 ELSE 0 END) as flow_percentage
FROM agent_interactions
WHERE timestamp > NOW() - INTERVAL '7 days'
GROUP BY agent_name;
